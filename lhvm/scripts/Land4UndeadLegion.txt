begin script GetTime

start

while QuestComplete != 1
	GL_Time = get game time

	if GL_Time < 5.99 or GL_Time > 18.01
		Day = GL_FALSE
		Night = GL_TRUE
	end if

	if GL_Time < 18.00 and GL_Time > 6.00
		Day = GL_TRUE
		Night = GL_FALSE
	end if
end while

end script GetTime
