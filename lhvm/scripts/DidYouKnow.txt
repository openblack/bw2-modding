//------------------------------------------------------------------------------------------------------------------------
//  DID YOU KNOW SCRIPTS
//  A specific script for bronze highlights
//------------------------------------------------------------------------------------------------------------------------
begin help script DidYouKnow(Location, HelpTextEnum, HelpTextCategory) 
	Highlight = 0
	AlreadyPlacedObj =  get HIGHLIGHT HIGHLIGHT_TIPS at {Location}
start
	if AlreadyPlacedObj not exists
		Highlight = create highlight CHALLENGE_NONE HIGHLIGHT_TIPS at {Location}
		set Highlight text property text constant HelpTextEnum category constant HelpTextCategory
		release Highlight
	end if
end script DidYouKnow

/*------------------------------------------------------------------------------------------------
	Script:		DeleteDYK
	Argument:	Position variable
	Function:	Deletes DidYouKnow at Position
------------------------------------------------------------------------------------------------*/
begin help script DeleteDYK(Location)
	DidYouKnow = 0
start
	DidYouKnow = get HIGHLIGHT SCRIPT_HIGHLIGHT_INFO_SCRIPT_BRONZE at {Location}
	delete DidYouKnow
end script DeleteDYK
